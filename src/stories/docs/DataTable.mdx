        {status}
      </p>

## Example

```js
<DataTable
  cellClass="whitespace-nowrap overflow-hidden overflow-ellipsis max-w-[150px]"
  rowsPerPageOptions={{ default: 3, options: [3, 5, 7] }}
  sortDescriptor={{ column: "service", direction: "ascending" }}
  inputSearch={{ variant: "bordered", color: "warning" }}
  selectionMode="multiple"
  onSelect={row => console.log(row)}
  loading={false}
  rows={[{name: "jhon doe", email: "jh@gmail.com" }]}
  columns={[{ key:"name", title: "first name" }]}
  localText={{
    emptyContent: "no data",
    items: ["user", "users"],
    paginateButtons: ["prev", "next"],
    rowsPerPage: "show rows",
  }}
  exportButton={{ color: "success", className: "text-white" }}
	onExport={rows => console.log(rows)}
/>
```
